Library MatchingLogic.Utils.wflexprod

From Coq Require Import ssreflect ssrfun ssrbool.

Section lexprod'.

Variable A B : Type.

Variable ltA : A → A → Prop.
Variable ltB : B → B → Prop.

Inductive lexprod' : A × B → A × B → Prop :=
| left_lex' : ∀ (x1 x2 : A) (y1 y2 : B), ltA x1 x2 → lexprod' (x1, y1) (x2, y2)
| right_lex' : ∀ (x : A) (y1 y2 : B), ltB y1 y2 → lexprod' (x, y1) (x, y2).

Lemma lexprod'_Acc :
  well_founded ltA →
  well_founded ltB →
  ∀ x, Acc ltA x → ∀ y, Acc ltB y → Acc lexprod' (x, y).

Theorem wf_lexprod' : well_founded ltA → well_founded ltB → well_founded lexprod'.

End lexprod'.
