  Lemma definedness_model_application :
    ∀ (M : @Model Σ) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),
      M ⊨ᵀ theory →
      ∀ (m: Domain M),
                 (app_ext (pattern_interpretation ρₑ ρₛ (sym definedness)) {[m]}) = ⊤.

        (@pattern_interpretation Σ M ρₑ ρₛ ϕ) ≠ ∅ →
        (@pattern_interpretation Σ M ρₑ ρₛ ⌈ ϕ ⌉ ) = ⊤.

  Lemma equal_iff_both_subseteq : ∀ (M : @Model Σ),
      M ⊨ᵀ theory →
      ∀ (ϕ1 ϕ2 : Pattern) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),
        @pattern_interpretation Σ M ρₑ ρₛ (ϕ1 =ml ϕ2) = ⊤ ↔
        (
          @pattern_interpretation Σ M ρₑ ρₛ (ϕ1 ⊆ml ϕ2) = ⊤ ∧
          @pattern_interpretation Σ M ρₑ ρₛ (patt_subseteq ϕ2 ϕ1) = ⊤).

  Lemma subseteq_iff_interpr_subseteq : ∀ (M : @Model Σ),
      M ⊨ᵀ theory →
      ∀ (ϕ1 ϕ2 : Pattern) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),
        @pattern_interpretation Σ M ρₑ ρₛ (ϕ1 ⊆ml ϕ2) = ⊤ ↔
        (@pattern_interpretation Σ M ρₑ ρₛ ϕ1)
          ⊆ (@pattern_interpretation Σ M ρₑ ρₛ ϕ2).

  Lemma equal_iff_interpr_same : ∀ (M : @Model Σ),
      M ⊨ᵀ theory →
      ∀ (ϕ1 ϕ2 : Pattern) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),
        @pattern_interpretation Σ M ρₑ ρₛ (ϕ1 =ml ϕ2) = ⊤ ↔
        @pattern_interpretation Σ M ρₑ ρₛ ϕ1
        = @pattern_interpretation Σ M ρₑ ρₛ ϕ2.

  Lemma equal_refl : ∀ (M : @Model Σ),
      M ⊨ᵀ theory →
      ∀ (ϕ : Pattern) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),
        @pattern_interpretation Σ M ρₑ ρₛ (patt_equal ϕ ϕ) = ⊤.

  Lemma equal_sym : ∀ (M : @Model Σ),
      M ⊨ᵀ theory →
      ∀ (ϕ1 ϕ2 : Pattern) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),
        @pattern_interpretation Σ M ρₑ ρₛ (ϕ1 =ml ϕ2) = ⊤ →
        @pattern_interpretation Σ M ρₑ ρₛ (patt_equal ϕ2 ϕ1) = ⊤.

  Lemma equal_trans : ∀ (M : @Model Σ),
      M ⊨ᵀ theory →
      ∀ (ϕ1 ϕ2 ϕ3 : Pattern) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),
        @pattern_interpretation Σ M ρₑ ρₛ (ϕ1 =ml ϕ2) = ⊤ →
        @pattern_interpretation Σ M ρₑ ρₛ (patt_equal ϕ2 ϕ3) = ⊤ →
        @pattern_interpretation Σ M ρₑ ρₛ (patt_equal ϕ1 ϕ3) = ⊤.

  Lemma free_evar_in_patt : ∀ (M : @Model Σ),

      ∀ (x : evar)(ϕ : Pattern) (ρₑ : @EVarVal Σ M) (ρₛ : @SVarVal Σ M),

        @pattern_interpretation Σ M ρₑ ρₛ (patt_in (patt_free_evar x) ϕ) = ⊤.

  Lemma single_element_definedness_impl_satisfies_definedness (M : @Model Σ) :
    (∃ (hashdef : Domain M),
        sym_interp M (inj definedness) = {[hashdef]}
        ∧ ∀ x, app_interp hashdef x = ⊤
    ) →
        satisfies_model M (axiom AxDefinedness).

  Lemma satisfies_definedness_implies_has_element_for_every_element
    (M : @Model Σ):
    M ⊨ᵀ theory →
    ∀ (x y : Domain M),
      ∃ (z : Domain M),
        z ∈ sym_interp M (inj definedness)
        ∧ y ∈ app_interp z x.
