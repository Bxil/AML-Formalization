  Let sym (s : Symbols) : Pattern :=
    @patt_sym Σ (inj s).

  Lemma definedness_model_application :
    ∀ (M : @Model Σ) (evar_val : @EVarVal Σ M) (svar_val : @SVarVal (Σ) M),
      M ⊨ᵀ theory →
      ∀ (m: Domain M),
                 (app_ext (pattern_interpretation evar_val svar_val (sym definedness)) {[m]}) = ⊤.

  Lemma definedness_not_empty_1 : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        (@pattern_interpretation (Σ) M evar_val svar_val phi) ≠ ∅ →
        (@pattern_interpretation (Σ) M evar_val svar_val (patt_defined phi)) = ⊤.

  Lemma definedness_empty_1 : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation Σ M evar_val svar_val phi = Semantics.Empty →
        @pattern_interpretation Σ M evar_val svar_val (patt_defined phi) = Semantics.Empty.

  Theorem modus_tollens: ∀ (P Q : Prop), (P → Q) → ¬Q → ¬P.

  Lemma definedness_empty_2 : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation Σ M evar_val svar_val (patt_defined phi) = Semantics.Empty →
        @pattern_interpretation Σ M evar_val svar_val phi = Semantics.Empty.

  Lemma definedness_not_empty_2 : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_defined phi) = Full →
        @pattern_interpretation (Σ) M evar_val svar_val phi ≠ Semantics.Empty.

  Lemma totality_not_full : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val phi ≠ ⊤ →
        @pattern_interpretation (Σ) M evar_val svar_val (patt_total phi) = ∅.

  Lemma totality_full : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val phi = ⊤ →
        @pattern_interpretation (Σ) M evar_val svar_val (patt_total phi) = ⊤.

  Lemma totality_result_empty : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_total phi) = ∅ →
        @pattern_interpretation (Σ) M evar_val svar_val phi ≠ ⊤.

  Lemma totality_result_nonempty : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_total phi) ≠ ∅ →
        @pattern_interpretation (Σ) M evar_val svar_val phi = ⊤.

  Lemma equal_iff_both_subseteq : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi1 phi2 : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi1 phi2) = ⊤ ↔
        (
          @pattern_interpretation (Σ) M evar_val svar_val (patt_subseteq phi1 phi2) = ⊤ ∧
          @pattern_interpretation (Σ) M evar_val svar_val (patt_subseteq phi2 phi1) = ⊤).

  Lemma subseteq_iff_interpr_subseteq : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi1 phi2 : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_subseteq phi1 phi2) = ⊤ ↔
        (@pattern_interpretation (Σ) M evar_val svar_val phi1)
          ⊆ (@pattern_interpretation (Σ) M evar_val svar_val phi2).

  Lemma equal_iff_interpr_same : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi1 phi2 : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi1 phi2) = Full ↔
        @pattern_interpretation (Σ) M evar_val svar_val phi1
        = @pattern_interpretation (Σ) M evar_val svar_val phi2.

  Lemma equal_refl : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi phi) = Full.

  Lemma equal_sym : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi1 phi2 : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi1 phi2) = Full →
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi2 phi1) = Full.

  Lemma equal_trans : ∀ (M : @Model (Σ)),
      M ⊨ᵀ theory →
      ∀ (phi1 phi2 phi3 : Pattern) (evar_val : @EVarVal (Σ) M) (svar_val : @SVarVal (Σ) M),
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi1 phi2) = Full →
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi2 phi3) = Full →
        @pattern_interpretation (Σ) M evar_val svar_val (patt_equal phi1 phi3) = Full.

  Lemma free_evar_in_patt : ∀ (M : @Model Σ),
      M ⊨ᵀ theory →
      ∀ (x : evar)(phi : Pattern) (evar_val : @EVarVal Σ M) (svar_val : @SVarVal Σ M),
        ((evar_val x) ∈ (@pattern_interpretation Σ M evar_val svar_val phi)) ↔
        @pattern_interpretation Σ M evar_val svar_val (patt_in (patt_free_evar x) phi) = ⊤.

  Lemma pattern_interpretation_eq_inversion_of ϕ₁ ϕ₂ M ρₑ ρₛ :
    M ⊨ᵀ theory →
    @pattern_interpretation Σ M ρₑ ρₛ (patt_eq_inversion_of ϕ₁ ϕ₂) = ⊤
    ↔ (∀ m₁ m₂,
            m₂ ∈ rel_of ρₑ ρₛ ϕ₁ m₁ ↔ m₁ ∈ rel_of ρₑ ρₛ ϕ₂ m₂
        ).

  Lemma single_element_definedness_impl_satisfies_definedness (M : @Model Σ) :
    (∃ (hashdef : Domain M),
        sym_interp M (inj definedness) = {[hashdef]}
        ∧ ∀ x, app_interp hashdef x = ⊤
    ) →
        satisfies_model M (axiom AxDefinedness).

  Lemma satisfies_definedness_implies_has_element_for_every_element
    (M : @Model Σ):
    M ⊨ᵀ theory →
    ∀ (x y : Domain M),
      ∃ (z : Domain M),
        z ∈ sym_interp M (inj definedness)
        ∧ y ∈ app_interp z x.

End definedness.
