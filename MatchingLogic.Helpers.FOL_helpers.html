Ltac wf_decompose_hypotheses :=
  unfold well_formed in × |- ;
  unfold well_formed_closed in × |- ;
  simpl in × |- ;
  repeat (
      match goal with
      | H : is_true _ |- _
        ⇒ unfold is_true in H
      | H : (andb _ _) = true |- _
        ⇒ apply andb_true_iff in H
      | H : (_ ∧ _) |- _
        ⇒ destruct H
      | H : (true = true) |- _
        ⇒ clear H
      end
    ).


Ltac wf_proof :=
  wf_decompose_hypotheses;
  unfold well_formed in *; unfold well_formed_closed in *; simpl;

  
  repeat (
      match goal with
      | H : (?X = true) |- context G[?X]
        ⇒ rewrite H
      end
    );

  simpl; reflexivity.
Ltac wf_check :=
match goal with
| |- well_formed _ ⇒ admit
| _ ⇒ fail
end.
